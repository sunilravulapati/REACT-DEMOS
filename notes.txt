import { useEffect, useState } from 'react'
import { useForm } from 'react-hook-form'
import { useNavigate } from 'react-router'

function ProductsList() {
  const { register, handleSubmit } = useForm();

  let [products, setProducts] = useState([])
  let [loading, setLoading] = useState(false)
  let [error, setError] = useState(null)
  let [productData, setProductData] = useState([]) //searched products data
  const navigate = useNavigate()

  // navigate to product component
  const gotoProduct = (productObj) => {
    // navigation logic
    // while navigating, transfer the product object as well
    navigate('/products', { state: { product: productObj } })
  }

  useEffect(() => {
    setLoading(true)
    async function getProducts() {
      try {
        let res = await fetch('https://fakestoreapi.com/products')
        if (res.status === 200) {
          // extract the json data from the response
          let productsData = await res.json()
          // update the state
          setProducts(productsData)
          setProductData(productsData)
        }
        else {
          throw new Error('failed to fetch the products data')
        }
      } catch (err) {
        setError(err.message)
      }
      finally {
        setLoading(false)
      }
    }
    getProducts()
  }, [])

  if (loading === true) {
    return <p className='text-center text-2xl text-blue-300'>Loading.....</p>
  }
  if (error !== null) {
    return <p className='text-center text-2xl text-red-500'>Error: {error}</p>
  }

  const searchProduct = (formDataObj) => {
    let searchTerm = formDataObj.searchInput.toLowerCase();
    if (searchTerm.trim() === '') {
      setProductData(products)
    }
    // console.log(searchTerm)
    let filtered = products.filter((productObj) => productObj.title.toLowerCase().includes(searchTerm))
    setProductData(filtered)
  }

  return (
    <div>
      {/* search by product category */}
      {/* search form field with a placeholder and a button */}
      {/* once the use clicks the button then implement a filter function to filter out by category and return the result */}
      <div className='flex justify-around'>
        <form onSubmit={handleSubmit(searchProduct)}>
          <input type="text" placeholder='enter the product' {...register('searchInput')} className='w-90 border mt-10 rounded' />
          <button className='ml-3 border rounded bg-gray-300 mt-10 '>Search 🔎</button>
        </form>
      </div>
      <div className='grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-12 mt-10'>
        {/* products list */}
        {
          productData.length === 0 ? (
            <p className='text-red-500 text-center'>
              No product found!
            </p>
          ) : (
            productData.map((productObj) => (
              <div
                onClick={() => gotoProduct(productObj)}
                key={productObj.id}
                className='shadow-md rounded-3xl p-10 text-center cursor-pointer bg-gray-100'
              >
                <img
                  className="h-40 object-contain block mx-auto mb-10"
                  src={productObj.image}
                  alt={productObj.title}
                />
                <h2>{productObj.title}</h2>
              </div>
            ))
          )
        }
      </div>
    </div>
  )
}

export default ProductsList

STATE MANAGEMENT
lets say there are 20 components all of them depending on state 'x' if the x is changed then the change needs to be broadcasted to all the components of the application
=>lifting state up / prop-drilling good for only single level one
->it means sharing the state across the application and achieve sync in state across the application
=>context api (solves the prop drilling) - small / medium applications
    we create a separate place named context provider and store the state(it exists outside the component tree) and the components will hence becomes context consumers
    now the components can directly take the state from the context provider
