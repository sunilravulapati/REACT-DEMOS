import { useState } from "react";
import AddTaskForm from "./AddTaskForm";
import TaskList from "./TaskList";

export default function TaskManager() {
  const [tasks, setTasks] = useState([]);

  // Calculate counts for the header
  const totalTasks = tasks.length;
  const completedTasks = tasks.filter((t) => t.completed).length;

  const addTask = (newTask) => {
    // Adding an object to an array state
    setTasks([...tasks, { ...newTask, id: Date.now(), completed: false }]);
  };

  const toggleTask = (id) => {
    setTasks(tasks.map(task => 
      task.id === id ? { ...task, completed: !task.completed } : task
    ));
  };

  const deleteTask = (id) => {
    setTasks(tasks.filter(task => task.id !== id));
  };

  return (
    <div className="max-w-2xl mx-auto p-6 bg-slate-50 min-h-screen">
      <h1 className="text-3xl font-bold text-slate-800 mb-2">Task Manager</h1>
      <div className="flex gap-4 mb-6 text-sm font-medium">
        <span className="bg-blue-100 text-blue-700 px-3 py-1 rounded-full">Total: {totalTasks}</span>
        <span className="bg-green-100 text-green-700 px-3 py-1 rounded-full">Completed: {completedTasks}</span>
      </div>

      <AddTaskForm onAddTask={addTask} />
      <TaskList tasks={tasks} onToggle={toggleTask} onDelete={deleteTask} />
    </div>
  );
}




import { useState } from "react";

export default function AddTaskForm({ onAddTask }) {
  const [title, setTitle] = useState("");
  const [priority, setPriority] = useState("Medium");
  const [error, setError] = useState("");

  const handleSubmit = (e) => {
    e.preventDefault();
    
    // Validation logic
    if (title.trim().length < 3) {
      setError("Title must be at least 3 characters long.");
      return;
    }

    onAddTask({ title, priority });
    
    // Resetting form (Phase 2 requirement)
    setTitle("");
    setPriority("Medium");
    setError("");
  };

  return (
    <form onSubmit={handleSubmit} className="bg-white p-4 rounded-xl shadow-sm border border-slate-200 mb-8">
      <div className="flex flex-col gap-3">
        <input
          type="text"
          placeholder="Task Title..."
          value={title}
          onChange={(e) => setTitle(e.target.value)}
          className="p-2 border rounded-md outline-blue-500"
        />
        {error && <p className="text-red-500 text-xs mt-1">{error}</p>}
        
        <select 
          value={priority} 
          onChange={(e) => setPriority(e.target.value)}
          className="p-2 border rounded-md"
        >
          <option>High</option>
          <option>Medium</option>
          <option>Low</option>
        </select>

        <button type="submit" className="bg-blue-600 text-white py-2 rounded-md hover:bg-blue-700 transition">
          Add Task
        </button>
      </div>
    </form>
  );
}


function TaskItem({ task, onToggle, onDelete }) {
  return (
    <div className={`flex items-center justify-between p-4 mb-3 rounded-lg border bg-white transition ${task.completed ? 'opacity-60 grayscale' : ''}`}>
      <div>
        <h3 className={`font-bold ${task.completed ? 'line-through' : ''}`}>{task.title}</h3>
        <p className="text-sm text-slate-500">Priority: {task.priority}</p>
      </div>
      <div className="flex gap-2">
        <button 
          onClick={() => onToggle(task.id)}
          className="px-3 py-1 text-sm bg-green-500 text-white rounded-md"
        >
          {task.completed ? "Undo" : "Done"}
        </button>
        <button 
          onClick={() => onDelete(task.id)}
          className="px-3 py-1 text-sm bg-red-100 text-red-600 rounded-md hover:bg-red-600 hover:text-white"
        >
          Delete
        </button>
      </div>
    </div>
  );
}

export default function TaskList({ tasks, onToggle, onDelete }) {
  return (
    <div className="mt-4">
      {tasks.length === 0 ? (
        <p className="text-center text-slate-400 mt-10">No tasks yet. Add one above!</p>
      ) : (
        tasks.map(task => (
          <TaskItem key={task.id} task={task} onToggle={onToggle} onDelete={onDelete} />
        ))
      )}
    </div>
  );
}
